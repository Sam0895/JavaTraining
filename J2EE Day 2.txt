// Sort policies by policy number
function sortPolicies() {
    // Example sorting logic
    // Replace with your actual sorting logic
    let policies = document.querySelectorAll('.policy');
    policies = Array.from(policies);
    policies.sort((a, b) => {
        let aNum = parseInt(a.dataset.policyNumber);
        let bNum = parseInt(b.dataset.policyNumber);
        return aNum - bNum;
    });

    // Clear current display
    let container = document.getElementById('policies');
    container.innerHTML = '';

    // Append sorted policies
    policies.forEach(policy => {
        container.appendChild(policy);
    });
}

// Display policy details in modal
function showPolicyDetails(policyNumber, policyHolder) {
    // Example: Display details in modal
    document.getElementById('policyNumber').textContent = policyNumber;
    document.getElementById('policyHolder').textContent = policyHolder;
    $('#policyModal').modal('show'); // Bootstrap modal display
}

// Example usage: Listen for document load
document.addEventListener('DOMContentLoaded', function() {
    // Add event listener for sorting button
    document.getElementById('sortButton').addEventListener('click', sortPolicies);

    // Add event listener for policy detail click
    let policies = document.querySelectorAll('.policy');
    policies.forEach(policy => {
        policy.addEventListener('click', function() {
            let policyNumber = this.dataset.policyNumber;
            let policyHolder = this.dataset.policyHolder;
            showPolicyDetails(policyNumber, policyHolder);
        });
    });
});
