import org.junit.After;
import org.junit.AfterClass;
import org.junit.Before;
import org.junit.BeforeClass;
import org.junit.Test;
import static org.junit.Assert.assertEquals;

public class MathOperationsTest {

    private MathOperations math;

    @BeforeClass
    public static void setUpClass() {
        System.out.println("Before Class: Setting up the test suite");
    }

    @AfterClass
    public static void tearDownClass() {
        System.out.println("After Class: Tearing down the test suite");
    }

    @Before
    public void setUp() {
        math = new MathOperations();
        System.out.println("Before: Setting up the test");
    }

    @After
    public void tearDown() {
        math = null;
        System.out.println("After: Tearing down the test");
    }

    @Test
    public void testAddition() {
        assertEquals(5, math.add(2, 3));
    }

    @Test
    public void testSubtraction() {
        assertEquals(1, math.subtract(3, 2));
    }

    @Test
    public void testMultiplication() {
        assertEquals(6, math.multiply(2, 3));
    }

    @Test
    public void testDivision() {
        assertEquals(3, math.divide(9, 3));
    }
}
